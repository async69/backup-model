import { createSlice } from '@reduxjs/toolkit'
import customStatus from '../../config/customStatus'
import { apiCallBegan } from '../api'
import endPoints from '../../config/endPoints'
 
export const initialState = {
    accounts: [],
    loading: false,
    status: customStatus.initial,
    errors: null,
    editStatus: {
        loading: false,
        status: customStatus.initial,
        errors: null
    },
    deleteStatus: {
        loading: false,
        status: customStatus.initial,
        errors: null
    },
    addStatus: {
        loading: false,
        status: customStatus.initial,
        errors: null
    }
}
 
export const name = 'chartsOfAccounts'
 
const accountSlice = createSlice({
    name,
    initialState,
    reducers: {
        accountsRequested(state, _) {
            state.loading = true
            state.status = customStatus.pending
            state.errors = null
        },
 
        accountsFetched(state, action) {
            state.loading = false
            state.status = customStatus.success
            state.errors = null
            state.accounts = action.payload
        },
 
        accountsRequestFailed(state, action) {
            state.loading = false
            state.status = customStatus.failed
            state.errors = action.payload
        },
 
        addAccountRequest(state, _) {
            state.addStatus.loading = true
            state.addStatus.status = customStatus.pending
            state.addStatus.errors = null
        },
 
        addAccountSuccess(state, action) {
            state.addStatus.loading = false
            state.addStatus.status = customStatus.success
            state.addStatus.errors = null
            state.accounts.push(action.payload)
        },
 
        addAccountRequestFailure(state, action) {
            state.addStatus.loading = false
            state.addStatus.status = customStatus.failed
            state.addStatus.errors = action.payload
        },
 
        editAccountRequested(state, _) {
            state.editStatus.loading = true
            state.editStatus.status = customStatus.pending
            state.editStatus.errors = null
        },
 
        editAccountSuccess(state, action) {
            state.editStatus.loading = false
            state.editStatus.status = customStatus.success

            const index = state.accounts.findIndex(account => account.id === action.payload.id)

        },
 
        editAccountRequestFailure(state, action) {
            state.editStatus.loading = false
            state.editStatus.status = customStatus.failed
            state.editStatus.errors = action.payload
        },
 
        deleteAccountRequested(state, _) {
            state.deleteStatus.loading = true
            state.deleteStatus.status = customStatus.pending
            state.deleteStatus.errors = null
        },
 
        deleteAccountSuccess(state, action) {

            state.deleteStatus.status = customStatus.success

        deleteAccountRequestFailed(state, action) {
            state.deleteStatus.loading = false
            state.deleteStatus.status = customStatus.failed
            state.deleteStatus.errors = action.payload
        }
    }
})
 
export const resolveState = state => ({
    entities: { [name]: state }
})
 
export const unResolveState = state => state.entities[name]
 
export const {
    accountsRequested, accountsFetched, accountsRequestFailed,
    addAccountRequest, addAccountSuccess, addAccountRequestFailure,
    editAccountRequested, editAccountSuccess, editAccountRequestFailure,
    deleteAccountRequested, deleteAccountSuccess, deleteAccountRequestFailed
} = accountSlice.actions
 
export const selectStatus = state => {
    const { loading, status, errors } = unResolveState(state)
    return {
        loading, status, errors
    }
}
 
export const selectEditStatus = state => {
    const { editStatus } = unResolveState(state)
    return editStatus
}
 
export const selectDeleteStatus = state => {
    const { deleteStatus } = unResolveState(state)
    return deleteStatus
}
 
export const selectAddStatus = state => {
    const { addStatus } = unResolveState(state)
    return addStatus
}
 
export const selectAccounts = state => {
    const { accounts } = unResolveState(state)
    return accounts
}

export const addAccount = account =>
    apiCallBegan({
        url: endPoints.mocks.account,
        method: "post",
        data: account,
        onStart: addAccountRequest.type,
        onSuccess: addAccountSuccess.type,
        onError: addAccountRequestFailure.type
    })
 
export const editAccount = account =>
    apiCallBegan({
        url: endPoints.mocks.account + "/" + account.id,
        method: "put",
        data: account,
        onStart: editAccountRequested.type,
        onSuccess: editAccountSuccess.type,
        onError: editAccountRequestFailure.type
    })
