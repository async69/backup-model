import { createSlice } from '@reduxjs/toolkit'
import { apiCallBegan } from '../api'
import endPoints from '../../config/endPoints'
import customStatus from '../../config/customStatus'
 
export const initialState = {
    loading: false,
    status: "initial",
    errors: null,
    fiscalYear: {}
}
 
export const name = "fiscalYears"
 
const fiscalYearSlice = createSlice({
    name,
    initialState,
    reducers: {
        requestAddFiscalYear(state, _) {
            state.loading = true
            state.status = customStatus.pending
            state.errors = null
        },
 
        successAddFiscalYear(state, action) {
            state.loading = false
            state.status = customStatus.success
            state.errors = null
            state.fiscalYear = action.payload
        },
 
        failureAddFiscalYear(state, action) {
            state.loading = false
            state.status = customStatus.failed
            state.errors = action.payload
        }
    }
})
 
export const {
    requestAddFiscalYear, successAddFiscalYear, failureAddFiscalYear
} = fiscalYearSlice.actions
 
export const resolveState = state => ({ entities: { [name]: state } })
export const unResolveState = state => state.entities[name]
 
export const selectStatus = state => {
    const { loading, status, errors } = unResolveState(state)
    return { loading, status, errors }
}
export const selectFiscalYear = state => {
    const { fiscalYear } = unResolveState(state)
    return fiscalYear
}
 
export default fiscalYearSlice.reducer
 
export const addFiscalYear = data =>
    apiCallBegan({
        url: endPoints.mocks.fiscalYear,
        method: "post",
        data,
        onStart: requestAddFiscalYear.type,
        onSuccess: successAddFiscalYear.type,
        onError: failureAddFiscalYear.type
    })
