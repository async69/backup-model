        errors: null

    editStatus: {
        loading: false,

        loading: false,

        errors: null
    }
}
 
export const name = 'generalJournals'
 
const journalSlice = createSlice({

    reducers: {

        addJournalRequestFailure(state, action) {

            state.addStatus.status = customStatus.failed
            state.addStatus.errors = action.payload
        },

})
